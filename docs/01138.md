import React, { Component } from 'react'

class ErrorBoundary extends Component {

  }

  componentDidCatch(error, info) {
    this.setState({ hasError: true })
    console.group('componentDidCatch')
    console.error(error)
    console.info(info)
    console.groupEnd('componentDidCatch')
  }

  render() {

    }

  }
}
